        function startMainMusic() {
            if(!musicStarted) { playSound("main.mp3", true, true); musicStarted = true; }
        }

        async function initGame() {
            startMainMusic();
            document.getElementById('start-screen').style.display = 'none';
            try {
                const s = await navigator.mediaDevices.getUserMedia({ video: true });
                document.getElementById('webcam').srcObject = s;
                document.getElementById('video-container').style.display = 'block';
            } catch(e) {}
            
            setTimeout(() => {
                document.getElementById('game-ui').style.opacity = '1';
                startFlow();
            }, 1000); 
        }

        function playSound(file, loop=false, bgm=false) {
            if(!bgm && activeSFX) { activeSFX.pause(); }
            if(bgm && activeBGM) { activeBGM.pause(); }
            const a = new Audio("music/" + file);
            a.loop = loop;
            if(bgm) activeBGM = a; else activeSFX = a;
            a.play().catch(e=>{});
        }

        function startFlow() {
            canA = false;
            document.getElementById('retry-layer').style.display = 'none';
            document.querySelectorAll('.money-step').forEach(s=>s.classList.remove('active'));
            
            // 背景顏色邏輯：1-10題深藍，11-20題全黑
            if(cur >= 10) {
                document.body.classList.add('bg-black');
            } else {
                document.body.classList.remove('bg-black');
            }

            const cs = document.getElementById(`step-${cur}`);
            if(cs) {
                cs.classList.add('active');
                cs.scrollIntoView({behavior:'smooth', block:'center'});
            }

            const pop = document.getElementById('money-popup');
            pop.innerHTML = `第 ${cur+1} 題<br>$${moneyList[cur].v}`;
            pop.style.display = 'block';
            pop.style.animation = 'popIn 0.8s forwards';
            
            playSound("new_q.mp3");
            setTimeout(() => {
                pop.style.display = 'none';
                if(cur < 10) playSound("normal.mp3", true, true); 
                else playSound("hard.mp3", true, true);
                
                document.getElementById('q-txt').innerText = quizData[cur].q;
                let opts = [...quizData[cur].a];
                let correctStr = opts[0];
                opts.sort(() => Math.random() - 0.5);
                
                for(let i=0; i<4; i++) {
                    const el = document.getElementById(`opt-${i}`);
                    el.style.visibility = 'visible';
                    el.classList.remove('selected','correct','wrong');
                    document.getElementById(`o-${i}`).innerText = opts[i];
                    el.onclick = () => checkAns(i, opts[i] === correctStr, correctStr);
                }
                canA = true;
            }, 3000);
        }

        function checkAns(idx, isCorr, correctText) {
            if(!canA) return; canA = false;
            const options = document.querySelectorAll('.option');
            options[idx].classList.add('selected');
            
            setTimeout(() => {
                if(activeBGM) activeBGM.pause();
                if(isCorr) {
                    options[idx].classList.add('correct');
                    confetti({particleCount:150, spread:70, origin:{y:0.6}});
                    playSound("correct.mp3");
                    setTimeout(() => { 
                        cur++; 
                        if(cur < 20) startFlow(); 
                        else alert("恭喜過關！"); 
                    }, 6000); 
                } else {
                    options[idx].classList.add('wrong');
                    options.forEach(el => { 
                        if (el.querySelector('div').innerText === correctText) el.classList.add('correct'); 
                    });
                    playSound("wrong.mp3");
                    setTimeout(() => { document.getElementById('retry-layer').style.display = 'flex'; }, 3000);
                }
            }, 2000);
        }

        function continueAfterWrong() {
            document.querySelectorAll('.lifeline').forEach(l=>l.classList.remove('used'));
            cur++; 
            if(cur < 20) startFlow();
        }

        function use5050() {
            if(!canA || document.getElementById('l-50').classList.contains('used')) return;
            document.getElementById('l-50').classList.add('used');
            let count = 0;
            const correctText = quizData[cur].a[0]; 
            const options = Array.from(document.querySelectorAll('.option')).sort(() => Math.random() - 0.5);
            for(let el of options) {
                if(el.querySelector('div').innerText !== correctText && count < 2) {
                    el.style.visibility = 'hidden';
                    count++;
                }
            }
        }
