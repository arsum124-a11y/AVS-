        let cur = 0, canA = false, musicStarted = false;
        let activeBGM = null, activeSFX = null;

        // 生成金額大道
        const ladderEl = document.getElementById('money-ladder');
        moneyList.forEach((v, i) => {
            const d = document.createElement('div');
            d.className = 'money-step' + ((i+1)%5===0 ? ' safe' : '');
            d.id = `step-${i}`;
            d.innerHTML = `<span>${i+1}</span><span>$${v}</span>`;
            ladderEl.appendChild(d);
        });

        function startMainMusic() {
            if(!musicStarted) {
                playSound("main.mp3", true, true);
                musicStarted = true;
            }
        }

        async function initGame() {
            startMainMusic();
            document.getElementById('start-screen').style.display = 'none';
            try {
                const s = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" } });
                document.getElementById('webcam').srcObject = s;
                document.getElementById('video-container').style.display = 'block';
            } catch(e) {}
            
            // 延遲一段時間讓 main.mp3 渲染氣氛，再開始問題
            setTimeout(() => {
                document.getElementById('game-ui').style.opacity = '1';
                startFlow();
            }, 4000); 
        }

        function playSound(file, loop=false, bgm=false) {
            if(!bgm && activeSFX) { activeSFX.pause(); activeSFX.src=""; }
            if(bgm && activeBGM) { activeBGM.pause(); activeBGM.src=""; }
            const a = new Audio("music/" + file);
            a.loop = loop;
            if(bgm) activeBGM = a; else activeSFX = a;
            a.play().catch(e=>{});
        }

        function startFlow() {
            canA = false;
            document.getElementById('retry-layer').style.display = 'none';
            document.querySelectorAll('.money-step').forEach(s=>s.classList.remove('active'));
            const cs = document.getElementById(`step-${cur}`);
            cs.classList.add('active');
            cs.scrollIntoView({behavior:'smooth', block:'center'});

            const pop = document.getElementById('money-popup');
            pop.innerText = `第 ${cur+1} 題\n$${moneyList[cur]}`;
            pop.style.display = 'block';
            pop.style.animation = 'popIn 0.8s forwards';
            
            playSound("new_q.mp3");
            setTimeout(() => {
                pop.style.display = 'none';
                playSound(cur < 10 ? "normal.mp3" : "hard.mp3", true, true);
                document.getElementById('q-txt').innerText = quizData[cur].q;
                
                // 隨機打亂選項
                let opts = [...quizData[cur].a];
                let correctStr = opts[0];
                opts.sort(() => Math.random() - 0.5);
                
                for(let i=0; i<4; i++) {
                    const el = document.querySelectorAll('.option')[i];
                    el.style.visibility = 'visible';
                    el.classList.remove('selected','correct','wrong');
                    document.getElementById(`o-${i}`).innerText = opts[i];
                    el.onclick = () => checkAns(i, opts[i] === correctStr);
                }
                canA = true;
            }, 4000);
        }

        function checkAns(idx, isCorr) {
            if(!canA) return; canA = false;
            document.querySelectorAll('.option')[idx].classList.add('selected');
            
            setTimeout(() => {
                if(activeBGM) activeBGM.pause();
                if(isCorr) {
                    document.querySelectorAll('.option')[idx].classList.add('correct');
                    confetti({particleCount:100, spread:70, origin:{y:0.6}});
                    playSound("correct.mp3");
                    setTimeout(() => { 
                        cur++; 
                        if(cur < 20) startFlow(); 
                        else alert("恭喜你成為百萬富翁！");
                    }, 5000);
                } else {
                    document.querySelectorAll('.option')[idx].classList.add('wrong');
                    playSound("wrong.mp3");
                    setTimeout(() => { document.getElementById('retry-layer').style.display = 'flex'; }, 2000);
                }
            }, 2000);
        }

        function continueAfterWrong() {
            document.querySelectorAll('.lifeline').forEach(l=>l.classList.remove('used'));
            cur++; 
            if(cur < 20) startFlow();
        }

        function use5050() {
            if(!canA || document.getElementById('l-50').classList.contains('used')) return;
            document.getElementById('l-50').classList.add('used');
            let count = 0;
            const correctText = quizData[cur].a[0];
            document.querySelectorAll('.option').forEach(el => {
                const txt = el.querySelector('div').innerText;
                if(txt !== correctText && count < 2) {
                    el.style.visibility = 'hidden';
                    count++;
                }
            });
        }
        function useL(id) { document.getElementById(id).classList.add('used'); }
